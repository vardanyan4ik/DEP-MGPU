Jenkinsfile
pipeline {
    agent any
    environment {
        IMAGE_NAME = 'liquidity-app-ci'
        TARGET_PLATFORMS = 'linux/amd64,linux/arm64'
    }
    stages {
        stage('Checkout') {
            steps {
                checkout scm
            }
        }
        stage('Lint') {
            steps {
                sh 'python3 -m py_compile app/*.py'
            }
        }
        stage('Quality Gate: Multi-Arch Config') {
            steps {
                sh '''
                echo "[gate] TARGET_PLATFORMS=$TARGET_PLATFORMS"
                echo "$TARGET_PLATFORMS" | grep -q 'linux/amd64' || (echo 'missing linux/amd64' && exit 1)
                echo "$TARGET_PLATFORMS" | grep -q 'linux/arm64' || (echo 'missing linux/arm64' && exit 1)
                '''
            }
        }
        stage('Build Multi-Arch') {
            steps {
                sh '''
                docker buildx create --name ci_builder --use || docker buildx use ci_builder
                docker buildx inspect --bootstrap
                mkdir -p raw
                docker buildx build \
                  --platform $TARGET_PLATFORMS \
                  --build-arg APP_VERSION=${BUILD_NUMBER} \
                  -t ${IMAGE_NAME}:${BUILD_NUMBER} \
                  --output=type=oci,dest=raw/liquidity_multiarch_${BUILD_NUMBER}.tar \
                  app
                '''
            }
        }
        stage('Test Run') {
            steps {
                sh '''
                docker build -t ${IMAGE_NAME}:smoke --build-arg APP_VERSION=${BUILD_NUMBER} app
                CID=$(docker run -d -p 18000:8000 ${IMAGE_NAME}:smoke)
                sleep 5
                curl -fsS http://localhost:18000/health
                docker rm -f $CID
                '''
            }
        }
    }
    post {
        always {
            sh 'docker buildx rm ci_builder || true'
        }
    }
}

.github/workflows/ci.yml
name: Lab4 CI Variant 9

on:
  push:
    branches: ["main", "master"]
  pull_request:

jobs:
  pipeline:
    runs-on: ubuntu-latest
    env:
      IMAGE_NAME: liquidity-app-ci
      TARGET_PLATFORMS: linux/amd64,linux/arm64
    steps:
      - name: Checkout
        uses: actions/checkout@v4

      - name: Set up Python
        uses: actions/setup-python@v5
        with:
          python-version: "3.10"

      - name: Lint (syntax gate)
        run: |
          python -m py_compile app/*.py

      - name: Quality Gate - Multi-Arch platforms
        run: |
          echo "TARGET_PLATFORMS=$TARGET_PLATFORMS"
          echo "$TARGET_PLATFORMS" | grep -q 'linux/amd64'
          echo "$TARGET_PLATFORMS" | grep -q 'linux/arm64'

      - name: Set up QEMU
        uses: docker/setup-qemu-action@v3

      - name: Set up Docker Buildx
        uses: docker/setup-buildx-action@v3

      - name: Build Multi-Arch OCI artifact
        run: |
          mkdir -p raw
          docker buildx build \
            --platform "$TARGET_PLATFORMS" \
            --build-arg APP_VERSION=${{ github.run_number }} \
            -t ${IMAGE_NAME}:${{ github.run_number }} \
            --output=type=oci,dest=raw/liquidity_multiarch_${{ github.run_number }}.tar \
            app

      - name: Smoke test single-arch image
        run: |
          docker build -t ${IMAGE_NAME}:smoke --build-arg APP_VERSION=${{ github.run_number }} app
          cid=$(docker run -d -p 18000:8000 ${IMAGE_NAME}:smoke)
          sleep 5
          curl -fsS http://localhost:18000/health
          docker rm -f "$cid"
